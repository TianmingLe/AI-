# 环境光传感器 (ALS) 与 HTTP/REST 客户端实施计划

> **对于代理工具（Agent）：** 需要子技能支持：使用 superpowers:subagent-driven-development（推荐）或 superpowers:executing-plans 逐个任务地执行此计划。步骤使用复选框（`- [ ]`）语法进行跟踪。

**目标**：实现一个可动态加载的 HTTP 客户端以支持 RESTful 云端交互，以及一个通过 I2C 读取光照数据的 ALS 传感器。

**架构**：`comm::HttpClient` 使用 `dlopen` 动态加载 `libcurl` 并处理 JSON REST 请求。`perception::AlsSensor` 使用 `<linux/i2c-dev.h>` 并支持 Mock 降级。两者将通过 `EventBus` 集成，并在主数据管道中实现联动。

**技术栈**：C++17, Linux I2C, libcurl, 动态加载（`dlopen`）, Catch2。

```cpp
#include "perception/AlsSensor.h"
#include "core/LogManager.h"
#include <fcntl.h>
#include <unistd.h>
#include <sys/ioctl.h>

namespace perception {
    AlsSensor::AlsSensor(const std::string& i2c_device) : use_mock_(false), fd_(-1), mock_lux_(300.0f), mock_increasing_(true) {
        if (!initI2C(i2c_device)) {
            LOG_WARN("Failed to initialize I2C ALS on " + i2c_device + ", falling back to mock mode.");
            use_mock_ = true;
        }
    }

    AlsSensor::~AlsSensor() {
        if (fd_ != -1) close(fd_);
    }

    bool AlsSensor::initI2C(const std::string& device_path) {
        fd_ = open(device_path.c_str(), O_RDWR);
        if (fd_ < 0) return false;
        return true; 
    }

    float AlsSensor::readLux() {
        if (use_mock_) {
            if (mock_increasing_) {
                mock_lux_ += 150.5f;
                if (mock_lux_ > 5000.0f) mock_increasing_ = false;
            } else {
                mock_lux_ -= 150.5f;
                if (mock_lux_ < 300.0f) mock_increasing_ = true;
            }
            return mock_lux_;
        }
        return 400.0f; // 实际硬件读取
    }
}
```

```cpp
#pragma once
#include <string>

namespace comm {
    class HttpClient {
    public:
        HttpClient();
        ~HttpClient();

        std::string get(const std::string& url);
        std::string post(const std::string& url, const std::string& json_payload);

        bool isUsingMock() const { return use_mock_; }

    private:
        bool loadCurlLibrary();
        void unloadCurlLibrary();

        bool use_mock_;
        bool library_loaded_;
        void* lib_handle_;
        
        // Function pointers
        void* (*curl_easy_init_ptr_)();
        void (*curl_easy_cleanup_ptr_)(void*);
        int (*curl_easy_setopt_ptr_)(void*, int, ...);
        int (*curl_easy_perform_ptr_)(void*);
    };
}
```
在 `src/comm/HttpClient.cpp` 中：
```cpp
#include "comm/HttpClient.h"
#include "core/LogManager.h"
#include <dlfcn.h>

namespace comm {
    HttpClient::HttpClient() : use_mock_(false), library_loaded_(false), lib_handle_(nullptr) {
        if (!loadCurlLibrary()) {
            LOG_WARN("Failed to load libcurl.so, falling back to HttpClient mock mode.");
            use_mock_ = true;
        }
    }

    HttpClient::~HttpClient() {
        unloadCurlLibrary();
    }

    bool HttpClient::loadCurlLibrary() {
        lib_handle_ = dlopen("libcurl.so.4", RTLD_NOW);
        if (!lib_handle_) lib_handle_ = dlopen("libcurl.so", RTLD_NOW);
        if (!lib_handle_) return false;
        
        curl_easy_init_ptr_ = (void* (*)())dlsym(lib_handle_, "curl_easy_init");
        curl_easy_cleanup_ptr_ = (void (*)(void*))dlsym(lib_handle_, "curl_easy_cleanup");
        curl_easy_setopt_ptr_ = (int (*)(void*, int, ...))dlsym(lib_handle_, "curl_easy_setopt");
        curl_easy_perform_ptr_ = (int (*)(void*))dlsym(lib_handle_, "curl_easy_perform");
        
        if (!curl_easy_init_ptr_ || !curl_easy_cleanup_ptr_ || !curl_easy_setopt_ptr_ || !curl_easy_perform_ptr_) {
            unloadCurlLibrary();
            return false;
        }
        library_loaded_ = true;
        return true;
    }

    void HttpClient::unloadCurlLibrary() {
        if (lib_handle_) {
            dlclose(lib_handle_);
            lib_handle_ = nullptr;
            library_loaded_ = false;
        }
    }

    std::string HttpClient::get(const std::string& url) {
        if (use_mock_ || !library_loaded_) {
            LOG_DEBUG("[HTTP Mock GET] " + url);
            return "{\"status\":\"mock_success\"}";
        }
        LOG_WARN("HttpClient real GET stubbed. Using mock.");
        return "{\"status\":\"mock_success\"}";
    }

    std::string HttpClient::post(const std::string& url, const std::string& json_payload) {
        if (use_mock_ || !library_loaded_) {
            LOG_DEBUG("[HTTP Mock POST] " + url + " Payload: " + json_payload);
            return "{\"status\":\"mock_success\"}";
        }
        LOG_WARN("HttpClient real POST stubbed. Using mock.");
        return "{\"status\":\"mock_success\"}";
    }
}
```
