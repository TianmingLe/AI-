# 电源管理 (PMIC) 与 NTP 协议实施计划

> **对于代理工具（Agent）：** 需要子技能支持：使用 superpowers:subagent-driven-development（推荐）或 superpowers:executing-plans 逐个任务地执行此计划。步骤使用复选框（`- [ ]`）语法进行跟踪。

**目标**：实现一个通过 I2C 读取硬件状态的电源传感器 (PowerSensor)，以及一个通过原生 UDP Sockets 获取网络时间并具有 Mock 降级功能的 NTP 客户端。

**架构**：`perception::PowerSensor` 使用 `<linux/i2c-dev.h>` 并支持在找不到硬件时模拟电量衰减。`comm::NtpClient` 使用 POSIX sockets 和 `settimeofday`，如果因沙盒环境缺少修改时间权限，则记录偏差日志并优雅降级。所有模块通过 `EventBus` 进行集成。

**技术栈**：C++17, Linux I2C, POSIX Sockets (UDP), Catch2。

```cpp
#include "perception/PowerSensor.h"
#include "core/LogManager.h"
#include <fcntl.h>
#include <unistd.h>
#include <sys/ioctl.h>

namespace perception {
    PowerSensor::PowerSensor(const std::string& i2c_device) : use_mock_(false), fd_(-1), mock_soc_(100.0f) {
        if (!initI2C(i2c_device)) {
            LOG_WARN("Failed to initialize I2C PMIC on " + i2c_device + ", falling back to mock mode.");
            use_mock_ = true;
        }
    }

    PowerSensor::~PowerSensor() {
        if (fd_ != -1) close(fd_);
    }

    bool PowerSensor::initI2C(const std::string& device_path) {
        fd_ = open(device_path.c_str(), O_RDWR);
        if (fd_ < 0) return false;
        return true; // 实际场景中需要 ioctl 设置 I2C 从设备地址
    }

    PowerData PowerSensor::readBatteryStatus() {
        if (use_mock_) {
            mock_soc_ -= 0.05f; // 模拟电量缓慢下降
            if (mock_soc_ < 0.0f) mock_soc_ = 0.0f;
            int voltage = 3300 + static_cast<int>(mock_soc_ * 9.0f); // 模拟 3300mV - 4200mV
            return {voltage, 150, mock_soc_}; // 模拟 150mA 放电电流
        }
        // 真实的 I2C 寄存器读取逻辑
        return {4200, 100, 100.0f};
    }
}
```

```cpp
#pragma once
#include <string>

namespace comm {
    class NtpClient {
    public:
        NtpClient();
        ~NtpClient();

        bool syncTime(const std::string& ntp_server = "pool.ntp.org");

}
```
在 `src/comm/NtpClient.cpp` 中：
```cpp
#include "comm/NtpClient.h"
#include "core/LogManager.h"
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <netdb.h>
#include <unistd.h>
#include <cstring>
#include <sys/time.h>

namespace comm {
    NtpClient::NtpClient() {}
    NtpClient::~NtpClient() {}

    bool NtpClient::syncTime(const std::string& ntp_server) {
        int sockfd = socket(AF_INET, SOCK_DGRAM, IPPROTO_UDP);
        if (sockfd < 0) {
            LOG_ERROR("NtpClient: Failed to create UDP socket.");
            return false;
        }

        // Set 2-second timeout
        struct timeval tv;
        tv.tv_sec = 2;
        tv.tv_usec = 0;
        setsockopt(sockfd, SOL_SOCKET, SO_RCVTIMEO, &tv, sizeof(tv));

        struct hostent* server = gethostbyname(ntp_server.c_str());
        if (!server) {
            LOG_WARN("NtpClient: Failed to resolve hostname: " + ntp_server);
            close(sockfd);
            return false;
        }

        struct sockaddr_in serv_addr;
        std::memset(&serv_addr, 0, sizeof(serv_addr));
        serv_addr.sin_family = AF_INET;
        std::memcpy(&serv_addr.sin_addr.s_addr, server->h_addr, server->h_length);
        serv_addr.sin_port = htons(123); // NTP Port

        // Prepare NTP packet (48 bytes)
        unsigned char msg[48] = {0};
        msg[0] = 0x1B; // LI = 0, VN = 3, Mode = 3 (Client)

        if (sendto(sockfd, msg, sizeof(msg), 0, (struct sockaddr*)&serv_addr, sizeof(serv_addr)) < 0) {
            LOG_WARN("NtpClient: Failed to send UDP packet.");
            close(sockfd);
            return false;
        }

        socklen_t server_len = sizeof(serv_addr);
        if (recvfrom(sockfd, msg, sizeof(msg), 0, (struct sockaddr*)&serv_addr, &server_len) < 0) {
            LOG_WARN("NtpClient: Receive timeout or error from " + ntp_server);
            close(sockfd);
            return false;
        }

        // Extract transmit timestamp (seconds since 1900)
        uint32_t tx_tm_s = ntohl(*((uint32_t*)&msg[40]));
        time_t current_time = tx_tm_s - 2208988800U; // Convert from 1900 to 1970 epoch

        struct timeval new_time;
        new_time.tv_sec = current_time;
        new_time.tv_usec = 0;

        if (settimeofday(&new_time, nullptr) < 0) {
            LOG_WARN("NtpClient: Network time received, but failed to set system time (EPERM). Running in Mock/Sandboxed mode.");
        } else {
            LOG_INFO("NtpClient: System time successfully synchronized via " + ntp_server);
        }

        close(sockfd);
        return true;
    }
}
```
